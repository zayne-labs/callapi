This page documents the configuration options specific to `createFetchClient`. These options define the base behavior for all requests made with the created client instance.

## Common Base Configuration Examples

### Basic API Client Setup

```ts title="api-client.ts"
import { createFetchClient } from "@zayne-labs/callapi";

export const apiClient = createFetchClient({
	baseURL: "https://api.example.com",
	timeout: 10000,
	retryAttempts: 3,
	throwOnError: true,
});
```

### Development vs Production Configuration

```ts title="api-client.ts"
import { createFetchClient } from "@zayne-labs/callapi";

export const apiClient = createFetchClient({
	baseURL: process.env.NODE_ENV === "production" ? "https://api.example.com" : "http://localhost:3000",
	timeout: process.env.NODE_ENV === "production" ? 10000 : 30000,
	throwOnError: true,

	// Enable detailed logging in development
	...(process.env.NODE_ENV === "development" && {
		plugins: [loggerPlugin({ mode: "verbose" })],
	}),
});
```

### Schema-Based Client

```ts title="typed-client.ts"
import { createFetchClient } from "@zayne-labs/callapi";
import { defineSchema } from "@zayne-labs/callapi/utils";
import { z } from "zod";

const apiSchema = defineSchema({
	"/users": {
		data: z.array(
			z.object({
				id: z.number(),
				name: z.string(),
				email: z.string(),
			})
		),
	},
	"/users/:id": {
		data: z.object({
			id: z.number(),
			name: z.string(),
			email: z.string(),
		}),
		params: z.object({
			id: z.number(),
		}),
	},
});

export const typedClient = createFetchClient({
	baseURL: "https://api.example.com",
	schema: apiSchema,
	throwOnError: true,
});
```
